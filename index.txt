<script>
// ════════════════════════════════════════════
//  ⚙️  CONFIGURATION — EDIT THIS LINE ONLY

var SCRIPT_URL = "YOUR_GOOGLE_APPS_SCRIPT_URL";
// ════════════════════════════════════════════

var regCount = 0;

function toggleType(card){
  var cb = card.querySelector('input[type=checkbox]');
  // let the native checkbox toggle, just update card style
  setTimeout(function(){
    card.classList.toggle('active', cb.checked);
  }, 0);
}
function syncType(cb){
  cb.closest('.type-card').classList.toggle('active', cb.checked);
}
function toggleOther(cb){
  cb.closest('.type-card').classList.toggle('active', cb.checked);
  var row = document.getElementById('otherRow');
  row.className = 'other-row' + (cb.checked ? ' show' : '');
  if(!cb.checked) document.getElementById('otherText').value='';
}
function selectDate(card){
  document.querySelectorAll('.date-card').forEach(function(c){c.classList.remove('selected')});
  setTimeout(function(){card.classList.add('selected')},0);
}

function showErr(id,show){
  var el=document.getElementById('err-'+id);
  if(el) el.className='err'+(show?' show':'');
}
function setInvalid(id,invalid){
  var el=document.getElementById(id);
  if(el){el.classList.toggle('invalid',invalid);}
}

function validate(){
  var ok=true;
  // types
  var types=Array.from(document.querySelectorAll('.type-grid input[type=checkbox]:checked'));
  showErr('type',!types.length); if(!types.length) ok=false;
  // fields
  ['company','fullname','position','phone'].forEach(function(f){
    var v=document.getElementById(f).value.trim();
    showErr(f,!v); setInvalid(f,!v); if(!v) ok=false;
  });
  // email
  var em=document.getElementById('email').value.trim();
  var emOk=em && /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(em);
  showErr('email',!emOk); setInvalid('email',!emOk); if(!emOk) ok=false;
  // date
  var dt=document.querySelector('input[name="attendance"]:checked');
  showErr('date',!dt); if(!dt) ok=false;
  return ok;
}

function submitForm(){
  if(!validate()) return;
  var btn=document.getElementById('submitBtn');
  btn.disabled=true; btn.textContent='Submitting…';

  var types=Array.from(document.querySelectorAll('.type-grid input[type=checkbox]:checked'))
    .map(function(c){
      if(c.value==='Other'){var t=document.getElementById('otherText').value.trim(); return t?'Other: '+t:'Other';}
      return c.value;
    });

  var attendance=document.querySelector('input[name="attendance"]:checked').value;
  var payload={
    timestamp: new Date().toLocaleString('en-GB'),
    visitorType: types.join(', '),
    company: document.getElementById('company').value.trim(),
    fullname: document.getElementById('fullname').value.trim(),
    position: document.getElementById('position').value.trim(),
    phone: document.getElementById('phone').value.trim(),
    email: document.getElementById('email').value.trim(),
    attendance: attendance,
    referral: document.getElementById('referral').value.trim() || '—'
  };

  if(SCRIPT_URL === "YOUR_GOOGLE_APPS_SCRIPT_URL"){
    // Demo mode — no real submission
    setTimeout(function(){
      showSuccess(attendance);
      btn.disabled=false; btn.textContent='Submit Registration';
    },800);
    return;
  }

  fetch(SCRIPT_URL, {
    method:'POST',
    mode:'no-cors',
    headers:{'Content-Type':'application/json'},
    body:JSON.stringify(payload)
  })
  .then(function(){
    showSuccess(attendance);
  })
  .catch(function(){
    alert('Network error. Please check your connection and try again.');
  })
  .finally(function(){
    btn.disabled=false; btn.textContent='Submit Registration';
  });
}

function showSuccess(date){
  regCount++;
  document.getElementById('regCount').textContent = regCount + ' registration' + (regCount===1?'':'s') + ' submitted this session';
  document.getElementById('confirmedDate').textContent = date;
  document.getElementById('formSection').style.display='none';
  document.getElementById('successScreen').className='success-screen show';
  document.getElementById('configNotice') && (document.getElementById('configNotice').style.display='none');
  window.scrollTo({top:0,behavior:'smooth'});
}

function resetForm(){
  document.getElementById('formSection').style.display='block';
  document.getElementById('successScreen').className='success-screen';
  // clear
  document.querySelectorAll('input[type=text],input[type=email],input[type=tel]').forEach(function(i){i.value='';i.classList.remove('invalid')});
  document.querySelectorAll('input[type=checkbox]').forEach(function(c){c.checked=false});
  document.querySelectorAll('input[type=radio]').forEach(function(r){r.checked=false});
  document.querySelectorAll('.type-card').forEach(function(c){c.classList.remove('active')});
  document.querySelectorAll('.date-card').forEach(function(c){c.classList.remove('selected')});
  document.getElementById('otherRow').className='other-row';
  document.querySelectorAll('.err').forEach(function(e){e.className='err'});
}
